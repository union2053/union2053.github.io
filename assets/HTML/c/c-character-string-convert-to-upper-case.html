C Character String Convert to upper case

PreviousNext

toupper() function converts the whole string to uppercase.

It also defines a function that uses ispunct() to count the number of punctuation characters in a string. 

#include <stdio.h>
#include <string.h>
#include <ctype.h>
#define LIMIT 81
void ToUpper(char *);

int main(void)
{
    char line[LIMIT];
    char * find;

    puts("Please enter a line:");
    fgets(line, LIMIT, stdin);
    find = strchr(line, '\n');    // look for newline
    if (find)                     // if the address is not NULL,
        *find = '\0';             // place a null character there
    ToUpper(line);
    puts(line);
    return 0;
}

void ToUpper(char * str)
{
    while (*str)
    {
        *str = toupper(*str);
        str++;
    }
}
