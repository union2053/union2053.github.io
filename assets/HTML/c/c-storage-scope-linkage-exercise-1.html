Question 1

Write and test in a loop a function that returns the number of times it has been called. 


#include <stdio.h>

int calls = 0;

void foo(void);

int main(void)
{
  for (int i = 0; i < 10; ++i) {
    foo();
  }
}

void foo(void)
{
  printf("I was called %d time%s previously!\n", calls, calls == 1 ? "" : "s");
  ++calls;
}

Question 2

Write a program that generates a list of 100 random numbers in the range 1-10 in sorted decreasing order. 


#include <stdio.h>
#include <stdlib.h>
#include <time.h>

void print_array(int size, const int arr[size]);
void randomize_array(int size, int *);
void sort_array(int size, int arr[size]);

int main(void)
{
  int array[100];
  randomize_array(100, array);
  sort_array(100, array);
  print_array(100, array);

  return 0;
}

void print_array(int size, const int arr[size])
{
  putchar('[');
  for (int i = 0; i < size; ++i)
    printf("%d, ", arr[i]);
  printf("\b\b]\n");
}

void randomize_array(int size, int *arr)
{
  srand((unsigned int) time(0));
  for (int i = 0; i < size; ++i)
    arr[i] = (rand() % 10) + 1;
}

void sort_array(int size, int arr[size])
{
  int temp;
  for (int i = 0; i < size; ++i) {
    for (int j = i+1; j < size; ++j) {
      if (arr[j] > arr[i]) {
        temp = arr[j];
        arr[j] = arr[i];
        arr[i] = temp;
      }
    }
  }
}

Question 3

Write a program that generates 1,000 random numbers in the range 1-10.

Print how many times each number was produced.

Have the program do this for 10 different seed values. 


#include <stdio.h>
#include <stdlib.h>
#include <time.h>

void print_array_stat(int size, const int arr[size]);
void randomize_array(unsigned seed, int size, int *);

int main(void)
{
  int array[10][1000];
  for (int i = 0; i < 10; ++i)
    randomize_array(i, 1000, array[i]);
  printf("Seed |");
  for (int i = 1; i < 11; ++i)
    printf("#%-4d", i);
  printf("\n     |------------------------------------------------\n");
  for (int i = 0; i < 10; ++i) {
    printf("  %d  |", i);
    print_array_stat(1000, array[i]);
  }

  return 0;
}

void randomize_array(unsigned seed, int size, int *arr)
{
  srand(seed);
  for (int i = 0; i < size; ++i)
    arr[i] = (rand() % 10) + 1;
}

void print_array_stat(int size, const int arr[size])
{
  int count[11] = { 0 };
  for (int i = 0; i < size; ++i)
    ++count[arr[i]];
  for (int i = 1; i < 11; ++i)
    printf("%-5d", count[i]);
  printf("\b\n");
}

Question 4

Create the program produce output like the following: 

Enter the number of sets; enter q to stop: 18
How many sides and how many dice? 6 3
Here are 18 sets of 3 6-sided throws.
12 10 6 9 8 14 8 15 9 14 12 17 11 7 10
13   8 14
How many sets? Enter q to stop: q


#include <stdio.h>
#include <stdlib.h>
#include <time.h>

int roll_dice(int sides);

int main(void)
{
  int sets, sides, dice, sum;
  srand((unsigned int) time(0));
  printf("Enter the number of sets; enter q to stop: ");
  while (scanf("%d", &sets) == 1) {
    printf("How many sides and how many dice? ");
    scanf("%d%d", &sides, &dice);
    printf("Here are %d set%s of %d %d-sided throw%s.\n",
           sets, sets > 1 ? "s" : "", dice, sides, dice > 1 ? "s" : "");
    for (int i = 0; i < sets; ++i) {
      sum = 0;
      for (int i = 0; i < dice; ++i)
        sum += roll_dice(sides);
      printf("%4d", sum);
    }
    printf("\nHow many sets? Enter q to stop: ");
  }
  return 0;
}

int roll_dice(int sides)
{
  return (rand() % sides) + 1;
}
