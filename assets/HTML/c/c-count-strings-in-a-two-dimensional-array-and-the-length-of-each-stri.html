C Count strings in a two-dimensional array and the length of each string

PreviousNext

The following code finds the number of strings in a two-dimensional array and the length of each string: 

#include <stdio.h> 

int main(void) 
{ 
  char str[][70] =  { 
                        "This is a test.", 
                        "This is another test.", 
                        "Test case from demo2s.com." 
                    }; 
  unsigned int count = 0;                              // Length of a string 
  unsigned int strCount = sizeof(str)/sizeof(str[0]);  // Number of strings 
  printf("There are %u strings.\n", strCount); 

  // find the lengths of the strings 
  for(unsigned int i = 0 ; i < strCount ; ++i) 
  { 
    count = 0; 
    while (str[i][count]) 
      ++count; 

    printf("The string:\n\"%s\"\n contains %u characters.\n", str[i], count); 
  } 

  return 0; 
}
