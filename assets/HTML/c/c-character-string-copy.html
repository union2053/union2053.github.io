strcpy() and strncpy() Functions

If pts1 and pts2 are both pointers to strings, the expression 

pts2 = pts1;

copies only the address of a string, not the string itself.

To copy a string, use the strcpy() function.

The strcpy() function is the string equivalent of the assignment operator. 

#include <stdio.h>
#include <string.h> // declares strcpy()
#define SIZE 40
#define LIM 5

int main(void)
{
    char qwords[LIM][SIZE];
    char temp[SIZE] = "test test test";
    int i = 0;

    strcpy(qwords[0], temp);
    strcpy(qwords[1], temp);
    strcpy(qwords[2], temp);
    strcpy(qwords[3], temp);
    strcpy(qwords[4], temp);
    
    for (i = 0; i < LIM; i++){
        puts(qwords[i]);
    }
    return 0;
}

strcpy() function is type char *.

It returns the value of its first argument-the address of a character.

The first argument need not point to the beginning of an array; this lets you copy just part of an array. 

#include <stdio.h>
#include <string.h>    // declares strcpy()
#define WORDS "demo"
#define SIZE 40

int main(void)
{
    const char * orig = WORDS;
    char copy[SIZE] = "This is a test from demo2s.com";
    char * ps;

    puts(orig);
    puts(copy);
    ps = strcpy(copy + 7, orig);
    puts(copy);
    puts(ps);

    return 0;
}

Output:

strcpy() copies the null character from the source string.

strncpy()

strcpy() function does not check to see whether the source string actually fits in the target string.

The safer way to copy strings is to use strncpy().

It takes a third argument, which is the maximum number of characters to copy.

The following code uses strncpy() instead of strcpy(). 

#include <stdio.h>
#include <string.h> /* declares strncpy() */
#define SIZE 40
#define TARGSIZE 7
#define LIM 5

int main(void)
{
    char qwords[LIM][TARGSIZE];
    char temp[SIZE]="this is a test from demo2s.com";
    int i = 0;

    strncpy(qwords[0], temp, TARGSIZE - 1);
    strncpy(qwords[1], temp, TARGSIZE - 1);
    strncpy(qwords[2], temp, TARGSIZE - 1);
    strncpy(qwords[3], temp, TARGSIZE - 1);
    strncpy(qwords[4], temp, TARGSIZE - 1);
    
    qwords[i][TARGSIZE - 1] = '\0';

    for (i = 0; i < LIM; i++)
        puts(qwords[i]);

    return 0;
}
