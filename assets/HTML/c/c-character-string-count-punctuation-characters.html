C Character String count punctuation characters

PreviousNext

The following code count punctuation characters in a String. 

#include <stdio.h>
#include <string.h>
#include <ctype.h>
#define LIMIT 81
int PunctCount(const char *);

int main(void)
{
    char line[LIMIT];
    char * find;

    puts("Please enter a line:");
    fgets(line, LIMIT, stdin);
    find = strchr(line, '\n');    // look for newline
    if (find)                     // if the address is not NULL,
        *find = '\0';             // place a null character there
    printf("That line has %d punctuation characters.\n",
           PunctCount(line));

    return 0;
}


int PunctCount(const char * str)
{
    int ct = 0;
    while (*str)
    {
        if (ispunct(*str))
            ct++;
        str++;
    }

    return ct;
}

Output:

The while (*str) loop processes each character in the string pointed to by str until the null character is reached.

At that point, the value of *str becomes 0 (the code for the null character), or false, and the loop terminates.
