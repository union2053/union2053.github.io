Since you are going through an array 0-7 of 1s, you should be using bit shifting:

int convertBinToDec(boolean Bin[]) {
  int ReturnInt = 0;
  for (int i = 0; i < 8; i++) {
    if (Bin[7 - i]) {
      Serial.print("Set Bit ");
      Serial.print(i);
      ReturnInt += 1<<i;
      Serial.print(" ==> ");
      Serial.print(1<<i);
      Serial.print(", ");
    }
  }
  return ReturnInt;
}
